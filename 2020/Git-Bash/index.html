Git

git init                        #初始化本地仓库.git
git status                      #显示当前所有文件的状态
git add File/.(all)             #用来将变动加到暂存区
git commit -m "info"            #提交变动至.git仓库
git log --oneline               #查看提交的版本信息
git reflog                      #查看全部的历史版本信息
git reset --hard 'number'       #通过版本号回退到某一个版本
git pull                        #从GitHub上拉取最新的代码
git push -f origin master       #将代码强制推送到master分支
-------------------------------------------------------------
git branch [Name]               #查看所有分支/新建分支
git checkout Name               #切换到某个分支
git merge Name                  #将某分支合并到当前分支
-------------------------------------------------------------
git remote add origin git@github.com:xxx/xxx.git
git push -u origin master       #将仓库上传到GitHub
